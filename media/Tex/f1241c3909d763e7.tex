def findPaths(int[][] grid):
    int m = grid.length
    int n = grid[0].length
    int[][] dp = new int[m][n]

    for (int c = n - 1; c >= 0; c--):
        dp[m-1][c] = 1
    
    for (int r = m - 1; r >= 0; r--):
        dp[r][n-1] = 1
    
    for (int r = m - 2; r >= 0; r--):
        for (int c = n - 2; c >= 0; c--):
            dp[r][c] = dp[r+1][c] + dp[r][c+1]
    
    return dp[0][0]
